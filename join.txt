1	SELECT * 
    FROM `students` 
    JOIN `degrees` ON `students`.`degree_id` = `degrees`.`id` 
    WHERE `degrees`.`name` = 'Corso di Laurea in Economia';

2	SELECT * 
    FROM `degrees` 
    JOIN `departments` ON `degrees`.`department_id` = `departments`.`id` 
    WHERE `departments`.`name` = 'Dipartimento di Neuroscienze';

3	SELECT * 
    FROM `courses` 
    JOIN `course_teacher` ON `courses`.`id` = `course_teacher`.`course_id` 
    WHERE `course_teacher`.`teacher_id` = 44;

4	SELECT * 
    FROM `students` JOIN `degrees` ON `students`.`degree_id` = `degrees`.`id` 
    JOIN `departments` ON `degrees`.`department_id` = `departments`.`id` 
    ORDER BY `students`.`surname`, `students`.`name`;

5	SELECT `degrees`.`name`, `teachers`.`name`, `teachers`.`surname`,`courses`.`name` 
    FROM `degrees` JOIN `courses` ON `courses`.`degree_id` = `degrees`.`id` 
    JOIN `course_teacher` ON `course_teacher`.`course_id` = `courses`.`id` 
    JOIN `teachers` on `course_teacher`.`teacher_id` = `teachers`.`id`;

6	SELECT DISTINCT `teachers`.`name`, `teachers`.`surname` 
    FROM `departments` JOIN `degrees` on `degrees`.`department_id` = `departments`.`id` 
    JOIN `courses` on `courses`.`degree_id` = `degrees`.`id` 
    JOIN `course_teacher` on `course_teacher`.`course_id` = `courses`.`id` 
    JOIN `teachers` on `course_teacher`.`teacher_id` = `teachers`.`id` 
    WHERE `departments`.`id` = 5 ORDER BY `teachers`.`surname` ASC;
